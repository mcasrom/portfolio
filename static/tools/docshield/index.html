#!/usr/bin/env python3 # -*- coding: utf-8 -*- """ DocShield — Marca de agua para documentos de identidad Procesado 100% local · RGPD-compliant Autor: tu-nombre | Licencia: MIT """ import os, sys, uuid, argparse from datetime import datetime from pathlib import Path from PIL import Image, ImageDraw, ImageFont try: import piexif HAS_PIEXIF = True except ImportError: HAS_PIEXIF = False def strip_exif(img: Image.Image) -> Image.Image: """Elimina todos los metadatos EXIF de la imagen.""" data = img.tobytes() clean = Image.frombytes(img.mode, img.size, data) clean.info = {} return clean def get_font(size: int) -> ImageFont.ImageFont: """Intenta cargar fuente del sistema, fallback a default.""" candidates = [ "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf", "/System/Library/Fonts/Helvetica.ttc", "C:/Windows/Fonts/arial.ttf", ] for path in candidates: if os.path.exists(path): try: return ImageFont.truetype(path, size) except: pass return ImageFont.load_default() def apply_watermark( img: Image.Image, recipient: str, purpose: str, valid_date: str = "", color: tuple = (255, 0, 80), opacity: int = 80, # 0-255 font_size: int = 14, style: str = "diagonal", add_uuid: bool = True, add_timestamp: bool = True, ) -> Image.Image: """ Aplica marca de agua multi-capa a documento de identidad. Args: img: Imagen PIL de entrada recipient: Destinatario del documento (entidad/persona) purpose: Finalidad del uso (texto libre) valid_date: Fecha de validez (YYYY-MM-DD) color: RGB color de la marca de agua opacity: Opacidad 0-255 (recomendado: 60-100) font_size: Tamaño de fuente en píxeles style: 'diagonal' | 'center' | 'grid' | 'corners' add_uuid: Añadir referencia única por documento add_timestamp: Añadir timestamp de generación Returns: Imagen PIL con marca de agua aplicada """ img = img.convert("RGBA") w, h = img.size # Capa transparente para la marca de agua overlay = Image.new("RGBA", (w, h), (0, 0, 0, 0)) draw = ImageDraw.Draw(overlay) font = get_font(font_size) font_small = get_font(max(9, font_size - 4)) ref_id = str(uuid.uuid4())[:8].upper() ts = datetime.now().strftime("%Y-%m-%d %H:%M") # Líneas de texto para la marca de agua lines = [ f"⚠ USO EXCLUSIVO: {recipient}", f"FINALIDAD: {purpose[:60]}", ] if valid_date: lines.append(f"VÁLIDO HASTA: {valid_date}") if add_uuid: lines.append(f"REF: {ref_id}") if add_timestamp: lines.append(f"GENERADO: {ts}") lines.append("NO REUTILIZAR · RGPD ART.5") text_block = " | ".join(lines) c = (*color, opacity) if style == "diagonal": # Tile diagonal: rotar texto y repetir tile_size = max(w, h) * 2 tile = Image.new("RGBA", (tile_size, tile_size), (0,0,0,0)) td = ImageDraw.Draw(tile) step = font_size * 6 for y in range(0, tile_size, step): td.text((0, y), text_block * 3, font=font, fill=c) rotated = tile.rotate(-35, expand=False) offset = ((w - tile_size) // 2, (h - tile_size) // 2) overlay.paste(rotated, offset, rotated) elif style == "center": # Marca central grande + borde main_font = get_font(font_size * 2) draw.text((w//2, h//2), lines[0], font=main_font, fill=c, anchor="mm") for i, line in enumerate(lines[1:]): draw.text((w//2, h//2 + (i+1)*font_size*2), line, font=font_small, fill=c, anchor="mm") # Marco de seguridad margin = 10 draw.rectangle([margin, margin, w-margin, h-margin], outline=c, width=3) elif style == "grid": step_x = w // 3 step_y = h // 4 for gx in range(0, w, step_x): for gy in range(0, h, step_y): for i, line in enumerate(lines): draw.text((gx, gy + i*font_size*1.2), line, font=font_small, fill=c) elif style == "corners": positions = [(10, 10), (w//2-50, h//2-20)] for px, py in positions: for i, line in enumerate(lines): draw.text((px, py + i * font_size * 1.4), line, font=font_small, fill=c) # Fusionar overlay result = Image.alpha_composite(img, overlay) return result.convert("RGB") def process_document( input_path: str, output_dir: str, recipient: str, purpose: str, **kwargs ) -> str: """Procesa un documento y guarda la versión protegida.""" src = Path(input_path) dst_dir = Path(output_dir) dst_dir.mkdir(parents=True, exist_ok=True) img = Image.open(src) # Strip EXIF si está disponible if kwargs.get("strip_exif", True): img = strip_exif(img) # Reducir resolución si anti_ocr=True if kwargs.get("anti_ocr", False): w, h = img.size img = img.resize((w//2, h//2), Image.LANCZOS) img = img.resize((w, h), Image.NEAREST) result = apply_watermark(img, recipient, purpose, **kwargs) ts_str = datetime.now().strftime("%Y%m%d_%H%M%S") out_name = f"PROTEGIDO_{src.stem}_{ts_str}.jpg" out_path = dst_dir / out_name result.save(out_path, "JPEG", quality=88, optimize=True) return str(out_path)

Marco Legal · Buenas Prácticas

¿Por qué es un riesgo enviar copias de DNI sin proteger?
El DNI/pasaporte contiene datos de categoría especial (nombre completo, fecha nacimiento, foto, número de documento). Su transmisión electrónica sin medidas técnicas puede vulnerar el RGPD Art. 5.1.f (integridad y confidencialidad) y el Art. 32 (medidas técnicas apropiadas). La AEPD ha sancionado casos similares.

¿Qué aporta la marca de agua jurídicamente?
Vincula el documento a un destinatario específico y finalidad declarada, creando evidencia de uso indebido si se reutiliza. No es una medida perfecta, pero cumple con el principio de accountability (responsabilidad proactiva) del RGPD.

Alternativas adicionales recomendadas
• Enviar por canales cifrados (correo con S/MIME o PGP, no WhatsApp)
• Solicitar confirmación de destrucción del documento tras el trámite
• Registrar en un log local qué copia se envió, a quién y cuándo
• Pedir a la entidad que acredite su base legal para solicitarlo

¿Puedes negarte a enviar una copia del DNI?
Sí. El RGPD contempla el principio de minimización de datos. Si la entidad puede verificar tu identidad por otros medios, tienes derecho a exigirlo. Si aun así lo solicitan, tienen obligación de informarte de la base legal y finalidad.

Reclamaciones y recursos
• AEPD (España): aepd.es — reclamaciones de protección de datos
• Tu DPO: Las empresas con más de 250 empleados deben tener un Delegado de Protección de Datos
• Formulario RGPD: puedes solicitar limitación del tratamiento, rectificación o supresión
